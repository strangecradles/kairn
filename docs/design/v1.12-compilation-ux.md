# v1.12.0 Design Doc — Compilation UX (Live Progress & Time Estimates)

> Make the multi-pass compilation feel alive. Show what's happening, estimate how long it'll take, and give the user something meaningful to watch instead of a frozen spinner.

**Status:** Planned  
**Depends on:** v1.11.0

---

## The Problem

After v1.10.1, complex compilations can take 2-7 minutes across two LLM passes. During that entire time, the user sees:

```
  ━━ Compilation ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  - Loading tool registry...
```

...and nothing else until it either succeeds or times out. Users think it's broken. This is especially bad for:
- First-time users who don't know what to expect
- Complex prompts (biotech, k8s, ML training) that take 5+ minutes
- Anyone who's used Claude Code and expects the subagent-style live status

---

## The Fix: Three UX Improvements

### 1. Phase-by-Phase Progress with Labels

Replace the single spinner with labeled phases that update as compilation proceeds:

```
  ━━ Compilation ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  ✔ Tool registry loaded (247 tools)
  ◐ Pass 1: Analyzing workflow & selecting tools...
```

Then:

```
  ✔ Tool registry loaded (247 tools)
  ✔ Pass 1: Selected 6 tools (Sequential Thinking, Context7, Exa, PostgreSQL, Memory, Security)
  ◐ Pass 2: Generating CLAUDE.md, commands, agents...
```

Then:

```
  ✔ Tool registry loaded (247 tools)
  ✔ Pass 1: Selected 6 tools — 8s
  ✔ Pass 2: Generated 5 commands, 3 agents, 2 rules — 42s
  ✔ Pass 3: Configured MCP servers & settings
  ✔ Environment compiled in 51s
```

If Pass 2 retries:

```
  ✔ Pass 1: Selected 6 tools — 8s
  ⚠ Pass 2: Response too large, retrying in concise mode...
  ◐ Pass 2 (retry): Generating environment...
```

### 2. Time Estimate

Show an estimate based on the compilation model and prompt complexity. This doesn't need to be precise — "~30s" or "~2-4 min" is enough to set expectations.

**Heuristic:** Estimate based on:
- Model speed tier (Haiku ~5s, Sonnet ~15-30s, Opus ~30-90s per pass)
- Number of passes (always 2 LLM calls + 1 deterministic)
- Prompt word count as a complexity proxy (>50 words = "complex")

```
  ━━ Compilation ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  Estimated time: ~45s (2 passes with claude-sonnet-4-6)

  ✔ Tool registry loaded (247 tools)
  ◐ Pass 1: Analyzing workflow & selecting tools...
```

For complex prompts:

```
  ━━ Compilation ━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━
  Estimated time: ~2-4 min (complex workflow, 2 passes with claude-sonnet-4-6)
```

### 3. Elapsed Timer

Show a running elapsed time next to the spinner so users know it's still working:

```
  ◐ Pass 2: Generating CLAUDE.md, commands, agents... [38s]
```

The `[38s]` updates every second. This alone eliminates the "is it frozen?" anxiety.

## Implementation

### Changes to `src/compiler/compile.ts`

The `compile()` function already has `onProgress?: (msg: string) => void`. Extend this to support richer status updates:

```typescript
interface CompileProgress {
  phase: 'registry' | 'pass1' | 'pass2' | 'pass2-retry' | 'pass3' | 'done';
  status: 'running' | 'success' | 'warning' | 'error';
  message: string;
  detail?: string;      // e.g. "Selected 6 tools"
  elapsed?: number;     // seconds since compilation started
  estimate?: string;    // e.g. "~45s"
}
```

Update `compile()` to emit structured progress:

```typescript
export async function compile(
  intent: string,
  onProgress?: (progress: CompileProgress) => void
): Promise<EnvironmentSpec> {
  const startTime = Date.now();
  const estimate = estimateTime(config.model, intent);
  
  onProgress?.({ phase: 'registry', status: 'running', message: 'Loading tool registry...', estimate });
  const registry = await loadRegistry();
  onProgress?.({ phase: 'registry', status: 'success', message: 'Tool registry loaded', detail: `${registry.length} tools` });

  // Pass 1
  onProgress?.({ phase: 'pass1', status: 'running', message: 'Analyzing workflow & selecting tools...' });
  const skeleton = await pass1(config, intent, registry);
  const toolNames = skeleton.tools.map(t => t.tool_id).join(', ');
  onProgress?.({ 
    phase: 'pass1', status: 'success', 
    message: `Selected ${skeleton.tools.length} tools`,
    detail: toolNames,
    elapsed: (Date.now() - startTime) / 1000
  });

  // Pass 2
  onProgress?.({ phase: 'pass2', status: 'running', message: 'Generating CLAUDE.md, commands, agents...' });
  let harness;
  try {
    harness = await pass2(config, intent, skeleton);
  } catch {
    onProgress?.({ phase: 'pass2-retry', status: 'warning', message: 'Response too large, retrying in concise mode...' });
    harness = await pass2(config, intent, skeleton, { concise: true });
  }
  const cmdCount = Object.keys(harness.commands).length;
  const agentCount = Object.keys(harness.agents ?? {}).length;
  const ruleCount = Object.keys(harness.rules).length;
  onProgress?.({ 
    phase: 'pass2', status: 'success',
    message: `Generated ${cmdCount} commands, ${agentCount} agents, ${ruleCount} rules`,
    elapsed: (Date.now() - startTime) / 1000
  });

  // Pass 3
  onProgress?.({ phase: 'pass3', status: 'running', message: 'Configuring MCP servers & settings...' });
  const settings = buildSettings(skeleton, registry);
  const mcpConfig = buildMcpConfig(skeleton, registry);
  onProgress?.({ phase: 'pass3', status: 'success', message: 'MCP servers & settings configured' });

  const totalElapsed = ((Date.now() - startTime) / 1000).toFixed(0);
  onProgress?.({ phase: 'done', status: 'success', message: `Environment compiled in ${totalElapsed}s` });

  // ... assemble and return spec
}
```

### Changes to `src/ui.ts`

Add the progress renderer:

```typescript
export function createProgressRenderer() {
  const lines: string[] = [];
  let intervalId: NodeJS.Timeout | null = null;
  let currentPhase = '';
  let phaseStart = Date.now();

  function render() {
    // Clear previous lines, rewrite all
    process.stdout.write('\x1B[' + (lines.length + 1) + 'A'); // move cursor up
    for (const line of lines) {
      process.stdout.write('\x1B[2K' + line + '\n'); // clear line + write
    }
    // Update elapsed on current running phase
    if (currentPhase) {
      const elapsed = Math.floor((Date.now() - phaseStart) / 1000);
      const lastIdx = lines.length - 1;
      lines[lastIdx] = lines[lastIdx].replace(/\[(\d+)s\]/, `[${elapsed}s]`);
    }
  }

  return {
    update(progress: CompileProgress) {
      if (progress.status === 'running') {
        currentPhase = progress.phase;
        phaseStart = Date.now();
        const est = progress.estimate ? `  (est. ${progress.estimate})` : '';
        lines.push(`  ◐ ${progress.message}${est} [0s]`);
        // Start elapsed timer
        if (!intervalId) {
          intervalId = setInterval(render, 1000);
        }
      } else if (progress.status === 'success') {
        // Replace the last "running" line with a success line
        const lastIdx = lines.length - 1;
        const elapsed = progress.elapsed ? ` — ${Math.floor(progress.elapsed)}s` : '';
        const detail = progress.detail ? ` (${progress.detail})` : '';
        lines[lastIdx] = `  ✔ ${progress.message}${detail}${elapsed}`;
        currentPhase = '';
      } else if (progress.status === 'warning') {
        const lastIdx = lines.length - 1;
        lines[lastIdx] = `  ⚠ ${progress.message}`;
        lines.push(`  ◐ Retrying... [0s]`);
      }
      render();
    },
    finish() {
      if (intervalId) clearInterval(intervalId);
      render();
    },
    fail(err: unknown) {
      if (intervalId) clearInterval(intervalId);
      const lastIdx = lines.length - 1;
      lines[lastIdx] = `  ✖ Compilation failed`;
      render();
    }
  };
}
```

### Estimate Heuristic

```typescript
function estimateTime(model: string, intent: string): string {
  const wordCount = intent.split(/\s+/).length;
  const isComplex = wordCount > 40;

  // Per-pass estimates by model tier
  const perPass: Record<string, number> = {
    'claude-3-5-haiku': 5,
    'claude-sonnet-4-6': 20,
    'claude-opus-4-6': 60,
    'gpt-4.1': 25,
    'gpt-4.1-mini': 10,
    'gemini-2.5-flash': 8,
    'gemini-2.5-pro': 30,
  };

  // Find closest match or default
  const basePerPass = Object.entries(perPass).find(([k]) => model.includes(k))?.[1] ?? 20;
  const totalBase = basePerPass * 2; // 2 LLM passes
  
  if (isComplex) {
    const low = Math.floor(totalBase * 1.5);
    const high = Math.floor(totalBase * 4);
    return `~${formatTime(low)}-${formatTime(high)} (complex workflow)`;
  }
  return `~${formatTime(totalBase)}`;
}

function formatTime(seconds: number): string {
  if (seconds < 60) return `${seconds}s`;
  const min = Math.floor(seconds / 60);
  const sec = seconds % 60;
  return sec > 0 ? `${min}m ${sec}s` : `${min} min`;
}
```
